5.3 Problem-solving success

plt_solutions <- function(data, problem) {
  data %>%
    filter(PROBLEM == problem) %>%
    ggplot(
      aes(
        x = EVAL_FIT_EST_MODE,
        y = solution_count,
        fill = EVAL_FIT_EST_MODE
      )
    ) +
    geom_col() +
    scale_y_continuous(
      limits = c(0, 30),
      breaks = seq(0, 30),
      labels = seq(0, 30)
    ) +
    scale_fill_bright() +
    scale_color_bright() +
    facet_grid(
       eval_mode_row ~ evals_per_gen
    ) +
    labs(title = problem) +
    theme(
      legend.position = "none",
      axis.text.x = element_text(angle = 45, hjust = 1)
    )
  ggsave(
    filename = paste0(plot_directory, problem, "-final.pdf"),
    height = 15,
    width = 7
  )
}

  comparison = character(),

for (g in groupings) {

  ft_results <- sol_stats_data %>%
    filter(grouping == g) %>%
    select(!grouping) %>%
    column_to_rownames(var = "EVAL_FIT_EST_MODE") %>%
    pairwise_fisher_test(
      p.adjust.method = "holm"
    ) %>%
    add_significance("p.adj")

  ft_results <- ft_results %>%
    mutate(
      comparison = rep(g, nrow(ft_results)),
      .keep = "all"
    ) %>%
    relocate(comparison)

  fisher_results <- rbind(
    fisher_results,
    ft_results
  )
}
